const scriptURL = 'https://script.google.com/macros/s/AKfycbz63uL-LUUGdu0M67uH9n-naBeE11khTwbteGi6WlJ04UEEGtx7fc8PVX3UtlaC9usj/exec'; const form = document.getElementById('hr-form'); const btn = document.getElementById('submit-btn'); form.addEventListener('submit', async e => { e.preventDefault(); btn.disabled = true; btn.innerText = 'שולח...'; const formData = new FormData(form); const dataObj = {}; // Собираем все данные из формы formData.forEach((value, key) => { if (key !== 'cv_file') dataObj[key] = value; }); const fileInput = document.getElementById('cv_file_input'); const sendToDatabase = async (payload) => { try { // Используем Content-Type: text/plain, чтобы избежать проблем с CORS await fetch(scriptURL, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'text/plain;charset=utf-8', }, body: JSON.stringify(payload) }); // В режиме no-cors мы не можем прочитать ответ, // поэтому считаем, что если запрос ушел без ошибки catch - всё ок alert("הנתונים התקבלו בהצלחה במערכת isradata.com!"); form.reset(); location.reload(); } catch (error) { console.error('Submission error:', error); alert("אירעה שגיאה בשליחה ל-isradata.com."); } finally { btn.disabled = false; btn.innerText = 'שליחת המועמדות'; } }; if (fileInput && fileInput.files.length > 0) { const file = fileInput.files[0]; const reader = new FileReader(); reader.onload = async () => { dataObj['fileData'] = reader.result; dataObj['fileName'] = file.name; await sendToDatabase(dataObj); }; reader.readAsDataURL(file); } else { await sendToDatabase(dataObj); } });
