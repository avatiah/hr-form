(שולח קובץ, אנא המתן)'; // Сообщаем, что идет загрузка файла const formData = new FormData(form); const dataObj = {}; // Собираем все текстовые поля formData.forEach((value, key) => { if (key !== 'cv_file') { dataObj[key] = value; } }); const fileInput = document.getElementById('cv_file_input'); // Функция для отправки данных const sendToServer = async (payload) => { try { const response = await fetch(scriptURL, { method: 'POST', mode: 'no-cors', // Важно для Google Script body: JSON.stringify(payload) }); // Так как mode: 'no-cors', мы не получим ответ JSON, // но если ошибок нет, значит данные ушли alert("הטופס נשלח בהצלחה!"); form.reset(); // Скрываем блоки document.getElementById('no-experience-block').classList.add('hidden'); document.getElementById('resume-options-block').classList.add('hidden'); } catch (error) { console.error('Error!', error); alert("אירעה שגיאה. בדוק את החיבור לאינטרנט."); } finally { btn.disabled = false; btn.innerText = 'שליחת המועמדות'; } }; // Если есть файл - читаем его, если нет - отправляем сразу if (fileInput && fileInput.files.length > 0) { const file = fileInput.files[0]; const reader = new FileReader(); reader.onload = async function() { dataObj['fileData'] = reader.result; dataObj['fileName'] = file.name; await sendToServer(dataObj); }; reader.readAsDataURL(file); } else { await sendToServer(dataObj); } });
